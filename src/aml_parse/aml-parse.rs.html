3use std::env;
4use std::error::Error;
5use std::fmt;
6use std::fs;
7use std::path::Path;
8use tartan_acpi::aml::parse::parse_table;
9

11fn main() -> Result<(), Box<dyn Error>> {
12    let args: Vec<_> = env::args().collect();
13    if args.len() <= 1 {
14        eprintln!("Usage: aml-parse INPUT");
15        return Err(Box::new(UsageError));
16    }
17
18    let aml_path = Path::new(&args[1]);
19    let aml_data = fs::read(aml_path)?;
20
21    match parse_table(&aml_data) {
22        Ok(t) => println!("Successfully parsed {}:\n{t:#x?}", aml_path.display()),
23        Err(e) => print!("Error parsing {}:\n\n{e}", aml_path.display()),
24    }
25
26    Ok(())
27}
