16/// File header for an ELF file with a specific address size.
17#[repr(C)]
18#[derive(Debug, Default, Clone, Copy, PartialEq, Eq)]
19pub struct Header<Addr: Copy> {
20    /// Header common to all ELF variants.
21    pub ident: HeaderIdent,
22    /// Distinguishes executables, libraries, etc.
23    pub file_type: FileType,
24    /// CPU architecture
25    pub machine: Machine,
26    /// ELF format version
27    pub version: u32,
28    /// Virtual address of executable entry point. Zero if not present.
29    pub entry_point: Addr,
30    /// File offset to the array of program section headers which control executable
31    /// loading. Zero if not present.
32    pub program_header_offset: Addr,
33    /// File offset to the array of section header which control linking. Zero if not
34    /// present.
35    pub section_header_offset: Addr,
36    /// CPU architecture-specific flags.
37    pub flags: u32,
38    /// Size in bytes of this header. May be longer than the structure in code.
39    pub header_size: u16,
40    /// Size in bytes of of each program segment header. May be longer than the structure
41    /// in code.
42    pub program_header_size: u16,
43    /// Number of program segment headers.
44    pub program_header_count: u16,
45    /// Size in bytes of each section header. May be longer than the structure in code.
46    pub section_header_size: u16,
47    /// Number of section headers.
48    pub section_header_count: u16,
49    /// Index of the section header which contains the list of section names.
50    pub section_names_index: u16,
51}
52
53const_assert_eq!(52, size_of::<Header<u32>>());
54const_assert_eq!(64, size_of::<Header<u64>>());
55#[cfg(target_pointer_width = "32")]
56const_assert_eq!(size_of::<HeaderNative>(), size_of::<Header<u32>>());
57#[cfg(target_pointer_width = "64")]
58const_assert_eq!(size_of::<HeaderNative>(), size_of::<Header<u64>>());
59
60impl<Addr: Copy> Header<Addr> {
61    /// Expected value of the [`version`](Self::version) field.
62    pub const VERSION: u32 = 1;
63
64    /// Checks that the header represents a valid ELF file for the current platform, and
65    /// panics otherwise
66    ///
67    /// # Panics
68    /// Panics when the header is not valid for the current platform.
69    pub fn verify_native(&self) {
70        self.ident.verify_native();
71        assert!(
72            self.version == Self::VERSION,
73            "Unsupported ELF header version {:#x}",
74            self.version,
75        );
76        assert!(
77            usize::from(self.header_size) >= size_of::<Self>(),
78            "Invalid ELF header size {:#x}. Expected at least {:#x} bytes.",
79            self.header_size,
80            size_of::<Self>(),
81        );
82        assert!(
83            usize::from(self.program_header_size) >= size_of::<ProgramHeaderNative>(),
84            "Invalid program header size {:#x}. Expected at least {:#x} bytes.",
85            self.program_header_size,
86            size_of::<ProgramHeaderNative>(),
87        );
88        assert!(
89            usize::from(self.section_header_size) >= size_of::<SectionHeaderNative>(),
90            "Invalid section header size {:#x}. Expected at least {:#x} bytes.",
91            self.section_header_size,
92            size_of::<SectionHeaderNative>(),
93        );
94        assert!(
95            self.machine == Machine::NATIVE,
96            "Unsupported ELF machine type {:?}",
97            self.machine,
98        );
99    }
100}

103/// Initial header which has the same layout in all ELF variants and determines how to
104/// interpret the rest of the file (endianness, sizes).
105#[repr(C)]
106#[derive(Debug, Default, Clone, Copy, PartialEq, Eq)]
107pub struct HeaderIdent {
108    /// Magic number that identifies this as an ELF file.
109    pub magic: [u8; 4],
110    /// Specifies the address size (32- or 64-bit) of the file.
111    pub class: FileClass,
112    /// Specifies whether the file uses big or little endian byte ordering.
113    pub endianness: Endianness,
114    /// Version of the ELF header.
115    pub header_version: u8,
116    /// Operating-system-specific ABI extensions that this file uses.
117    pub os_abi: OSABI,
118    /// Backwards-incompatible version number of the [`os_abi`](Self::os_abi) extensions
119    /// that this file uses.
120    pub os_abi_version: u8,
121    _pad: [u8; 7],
122}
123
124const_assert_eq!(16, size_of::<HeaderIdent>());
125
126impl HeaderIdent {
127    /// Expected value of the [`magic`](Self::magic) field.
128    pub const MAGIC: [u8; 4] = [0x7f, 0x45, 0x4c, 0x46];
129
130    /// Expected value of the [`header_version`](Self::header_version) field.
131    pub const VERSION: u8 = 1;
132
133    /// Checks that the header represents a valid ELF file, of any variant, and panics
134    /// otherwise.
135    ///
136    /// # Panics
137    /// Panics when the header is not valid.
138    pub fn verify_format(&self) {
139        assert!(
140            self.magic == Self::MAGIC,
141            "Unrecognized magic number for ELF file: {:#x?}",
142            self.magic,
143        );
144        assert!(
145            self.header_version == Self::VERSION,
146            "Unsupported ELF identity header version {:#x}",
147            self.header_version,
148        );
149    }
150
151    /// Checks that the header represents an ELF file in the native endianness and address
152    /// size, and panics otherwise.
153    ///
154    /// # Panics
155    /// Panics when the header is not valid for the current platform.
156    pub fn verify_native(&self) {
157        self.verify_format();
158        assert!(
159            self.endianness == Endianness::NATIVE,
160            "ELF variant {:?} not supported on this architecture",
161            self.endianness,
162        );
163        assert!(
164            self.class == FileClass::NATIVE,
165            "ELF variant {:?} not supported on this architecture",
166            self.class,
167        );
168    }
169}

457/// Header that controls linking
458#[repr(C)]
459#[derive(Debug, Default, Clone, Copy, PartialEq, Eq)]
460pub struct SectionHeader<Addr: Copy> {
461    /// Offset within the names section to the start of this section's name.
462    pub name_table_offset: u32,
463    /// Indicates the contents/meaning of the section described by this header.
464    pub section_type: SectionType,
465    /// Flags for various attributes of this section.
466    ///
467    /// This is the raw numeric value of the flags. Use the [`flags`](Self::flags)
468    /// accessor to view as a [`SectionFlags`] type.
469    pub flags_raw: Addr,
470    /// Virtual address where this section should be loaded in memory, if applicable.
471    pub virtual_addr: Addr,
472    /// File offset to the start of this section's data.
473    pub file_offset: Addr,
474    /// Size in bytes of this section's file data.
475    pub file_size: Addr,
476    /// Depending on the [`section_type`](Self::section_type), this may identify another
477    /// related section.
478    pub associated_section_index: u32,
479    /// Additional flags that depend on the [`section_type`](Self::section_type).
480    pub extra_info: u32,
481    /// Required alignment of the section in bytes.
482    pub alignment: Addr,
483    /// Depending on the [`section_type`](Self::section_type), this may identify the size
484    /// of individual data entries within the section.
485    pub entry_size: Addr,
486}
487
488const_assert_eq!(40, size_of::<SectionHeader<u32>>());
489const_assert_eq!(64, size_of::<SectionHeader<u64>>());
490#[cfg(target_pointer_width = "32")]
491const_assert_eq!(size_of::<SectionHeaderNative>(), size_of::<SectionHeader<u32>>());
492#[cfg(target_pointer_width = "64")]
493const_assert_eq!(size_of::<SectionHeaderNative>(), size_of::<SectionHeader<u64>>());
