10#[cfg(any(target_arch = "x86", target_arch = "x86_64"))]
11pub fn print_state(out: &mut dyn fmt::Write) -> fmt::Result {
12    use tartan_arch::x86_common::{self, features, interrupt, paging, protection};
13
14    writeln!(out, "CPUID max basic: {:x?}", features::max_cpuid_index_basic())?;
15    writeln!(out, "CPUID max ext.: {:x?}", features::max_cpuid_index_extended())?;
16
17    let basic_features = features::BasicFeatures::get();
18    writeln!(out, "{basic_features:#x?}")?;
19    writeln!(out, "{:#x?}", features::ExtendedFeatures::get())?;
20    writeln!(out, "{:#x?}", features::AddressSpaceSizes::get())?;
21
22    writeln!(out, "{:#x?}", x86_common::FlagRegister::get())?;
23    writeln!(out, "{:#x?}", x86_common::ControlRegister0::get())?;
24    writeln!(out, "{:#x?}", x86_common::ControlRegister4::get())?;
25
26    writeln!(out, "{:#x?}", paging::ControlRegister2::get())?;
27    writeln!(out, "{:#x?}", paging::ControlRegister3::get())?;
28
29    #[allow(clippy::if_not_else)]
30    if !basic_features.extended_state_save() {
31        writeln!(out, "ExtendedControlRegister0 unsupported")?;
32    } else if !basic_features.extended_state_save_enabled() {
33        writeln!(out, "ExtendedControlRegister0 disabled")?;
34    } else {
35        writeln!(out, "{:#x?}", x86_common::ExtendedControlRegister0::get())?;
36    }
37
38    #[cfg(target_arch = "x86_64")]
39    {
40        use tartan_arch::x86_64;
41
42        writeln!(out, "{:#x?}", x86_64::ControlRegister8::get())?;
43        writeln!(out, "{:#x?}", x86_64::ExtendedFeatureEnableRegister::get())?;
44    }
45
46    writeln!(out, "{:#x?}", interrupt::InterruptDescriptorTableRegister::get())?;
47    writeln!(out, "{:#x?}", protection::GlobalDescriptorTableRegister::get())?;
48    describe_segment_register(
49        out,
50        "LDTR",
51        protection::LocalDescriptorTableRegister::get(),
52    )?;
53    describe_segment_register(out, "TR", protection::TaskRegister::get())?;
54    describe_segment_register(out, "CS", protection::SegmentRegister::Code.get())?;
55    describe_segment_register(out, "DS", protection::SegmentRegister::Data.get())?;
56    describe_segment_register(out, "SS", protection::SegmentRegister::Stack.get())?;
57    describe_segment_register(out, "ES", protection::SegmentRegister::Extra.get())?;
58    describe_segment_register(out, "FS", protection::SegmentRegister::ExtraF.get())?;
59    describe_segment_register(out, "GS", protection::SegmentRegister::ExtraG.get())?;
60
61    Ok(())
62}

64#[cfg(any(target_arch = "x86", target_arch = "x86_64"))]
65fn describe_segment_register(
66    out: &mut dyn fmt::Write,
67    name: &str,
68    selector: tartan_arch::x86_common::protection::Selector,
69) -> fmt::Result {
70    use tartan_arch::x86_common::protection::{
71        DescriptorFlags, GateDescriptor, GenericDescriptor, SegmentDescriptor,
72    };
73
74    writeln!(out)?;
75    writeln!(out, "{name}:")?;
76    writeln!(out, "{selector:#x?}")?;
77
78    let descriptor = selector.descriptor_address() as *const GenericDescriptor;
79    let descriptor_flags = unsafe { (*descriptor).flags };
80    if descriptor_flags.is_gate() {
81        let gate_descriptor = unsafe { &*(descriptor as *const GateDescriptor) };
82        writeln!(out, "{:#x?}", *gate_descriptor)?;
83    } else {
84        let seg_descriptor = unsafe { &*(descriptor as *const SegmentDescriptor) };
85        writeln!(out, "{:#x?}", *seg_descriptor)?;
86    }
87
88    Ok(())
89}
