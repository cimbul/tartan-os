19/// Parse struct fields in order using an initializer-like syntax
20///
21/// # Example
22///
23/// ```
24/// # use nom::IResult;
25/// # use nom::character::complete::alphanumeric1;
26/// # use nom::number::complete::le_u16;
27/// # use tartan_parsers::struct_parser;
28/// #
29/// type Parser<'a, T> = fn(&'a [u8]) -> IResult<&'a [u8], T, ()>;
30///
31/// # #[derive(Debug, Clone, Copy, PartialEq, Eq)]
32/// struct Foo<'a> { a: u16, b: &'a [u8] };
33///
34/// // Will parse `a`, then `b`, and return the struct if both succeed
35/// let parse_foo: Parser<Foo> = struct_parser!(
36///     Foo {
37///         a: le_u16,
38///         b: alphanumeric1,
39///     }
40/// );
41///
42/// assert_eq!(
43///     parse_foo(b"\x34\x12Bar10"),
44///     Ok((b"" as &[u8], Foo { a: 0x1234, b: b"Bar10" }))
45/// );
46/// ```
47#[macro_export]
48macro_rules! struct_parser {
49    // Struct form
50    [
51        $struct:ident $( :: $struct_x:ident )* {
52            $( $field:ident : $parser:expr ),+
53            $(,)?
54        }
55    ] => {
56        |i| {
57            $( let (i, $field) = $parser(i)?; )+
58            let result = ($struct $(::$struct_x)* { $($field),+ });
59            Ok((i, result))
60        }
61    };
62
63    // Tuple form
64    [
65        $struct:ident $( :: $struct_x:ident )* (
66            $( $parser:expr ),+
67            $(,)?
68        )
69    ] => {
70        |i| {
71            let (i, t) = nom::sequence::tuple(( $($parser),+ ))(i)?;
72            let result = Fn::call(& $struct $(::$struct_x)*, t);
73            Ok((i, result))
74        }
75    };
76}

79/// Combinator for productions that use a deterministic opcode
80///
81/// If a production is preceded by an unambiguous opcode, then we can avoid
82/// backtracking when parsing the rest. This enables better error messages in addition
83/// to faster parsing.
84pub fn opcode<I, O1, O2, E, P, Q>(
85    description: &'static str,
86    opcode_parser: P,
87    body_parser: Q,
88) -> impl FnMut(I) -> IResult<I, O2, E>
89where
90    I: Clone,
91    P: Parser<I, O1, E>,
92    Q: Parser<I, O2, E>,
93    E: ParseError<I> + ContextError<I>,
94{
95    let mut parser = preceded(opcode_parser, cut(body_parser));
96    move |i: I| match parser.parse(i.clone()) {
97        Err(nom::Err::Failure(e)) => {
98            // Only add context to *failures*, on the assumption that these will only
99            // come from the opcode body, and we don't want to add context when the
100            // opcode itself wasn't recognized.
101            Err(nom::Err::Failure(E::add_context(i, description, e)))
102        }
103        other => other,
104    }
105}

108/// Create an iterator that repeatedly executes the parser.
109///
110/// Unlike [`nom::combinator::iterator`], this yields a [`Result`] at every step, so
111/// errors are returned in the stream, and there is no separate `finish()` method. If the
112/// parser encounters an error, the iterator will yield that error and not attempt to
113/// parse any more elements.
114pub fn result_iterator<P, I, O, E>(
115    input: I,
116    parser: P,
117) -> impl Iterator<Item = Result<O, nom::Err<E>>>
118where
119    P: Fn(I) -> IResult<I, O, E>,
120    I: InputLength,
121    E: ParseError<I>,
122{
123    ResultIterator { input: Some(input), parser }
124}
125
126
127struct ResultIterator<P, I, O, E>
128where
129    P: Fn(I) -> IResult<I, O, E>,
130    I: InputLength,
131    E: ParseError<I>,
132{
133    input: Option<I>,
134    parser: P,
135}
136
137impl<P, I, O, E> Iterator for ResultIterator<P, I, O, E>
138where
139    P: Fn(I) -> IResult<I, O, E>,
140    I: InputLength,
141    E: ParseError<I>,
142{
143    type Item = Result<O, nom::Err<E>>;
144
145    fn next(&mut self) -> Option<Self::Item> {
146        let mut input = None;
147        core::mem::swap(&mut self.input, &mut input);
148        match input {
149            None => None,
150            Some(i) if i.input_len() == 0 => None,
151            Some(i) => match (self.parser)(i) {
152                Ok((rest, output)) => {
153                    self.input = Some(rest);
154                    Some(Ok(output))
155                }
156                Err(e) => {
157                    self.input = None;
158                    Some(Err(e))
159                }
160            },
161        }
162    }
163}
