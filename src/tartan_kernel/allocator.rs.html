9/// Simple kernel allocator that keeps a singly-linked list of blocks and finds the first
10/// fit.
11pub struct Allocator<'a> {
12    block_list: Option<RefCell<BlockList<'a>>>,
13}
14
15impl<'a> Allocator<'a> {
16    pub const fn uninitialized() -> Self {
17        Self { block_list: None }
18    }
19
20    pub fn init(&mut self, block_list: BlockList<'a>) {
21        self.block_list = Some(RefCell::new(block_list));
22    }
23}
24
25unsafe impl GlobalAlloc for Allocator<'_> {
26    unsafe fn alloc(&self, layout: Layout) -> *mut u8 {
27        let layout = layout.align_to(align_of::<BlockHeader>()).unwrap();
28        let mut cursor = self.block_list.as_ref().unwrap().borrow_mut().front_mut();
29        while !cursor.current().is_end() {
30            if cursor.current().as_ref().can_fit(layout) {
31                cursor.split_to_align(layout);
32                cursor.split_to_minimum(layout.size());
33                return cursor.current().as_ref().data_ptr();
34            }
35            cursor.move_next();
36        }
37        core::ptr::null_mut()
38    }
39
40    unsafe fn dealloc(&self, ptr: *mut u8, _: Layout) {
41        let mut block_list = self.block_list.as_ref().unwrap().borrow_mut();
42        let mut cursor = block_list.cursor_from_data_ptr(ptr);
43        cursor.current().set_free(true);
44
45        // TODO: This would be much faster with a doubly-linked list
46        let mut cursor = block_list.front_mut();
47        while !cursor.current().is_end() {
48            cursor.merge_next_if_free();
49            cursor.move_next();
50        }
51    }
52
53    // TODO: Implement realloc
54}

57pub struct BlockList<'a> {
58    data: &'a mut [MaybeUninit<usize>],
59}
60
61impl<'a> BlockList<'a> {
62    pub fn from_block(data: &'a mut [MaybeUninit<usize>]) -> Self {
63        let mut start_header = BlockHeader::default();
64        start_header.set_data_size((data.len() - 2) * size_of::<BlockHeader>());
65        start_header.set_free(true);
66        data[0] = MaybeUninit::new(start_header.into());
67
68        let mut end_header = BlockHeader::default();
69        end_header.set_free(false);
70        data[data.len() - 1] = MaybeUninit::new(end_header.into());
71
72        Self { data }
73    }
74
75    unsafe fn cursor_from_data_ptr(&mut self, data_ptr: *mut u8) -> CursorMut<'a> {
76        let header_addr = data_ptr as usize - size_of::<BlockHeader>();
77        debug_assert!(header_addr.is_multiple_of(align_of::<BlockHeader>()));
78
79        let header_ptr = header_addr as *mut BlockHeader;
80        debug_assert!(self.data.as_ptr_range().contains(&(header_addr as *const _)));
81
82        CursorMut(Pin::new_unchecked(&mut *header_ptr))
83    }
84
85    #[cfg(test)]
86    fn front(&self) -> Cursor<'a> {
87        #![allow(clippy::transmute_ptr_to_ptr)]
88        Cursor(unsafe {
89            Pin::new_unchecked(transmute::<&core::mem::MaybeUninit<usize>, &BlockHeader>(
90                &self.data[0],
91            ))
92        })
93    }
94
95    fn front_mut(&mut self) -> CursorMut<'a> {
96        #![allow(clippy::transmute_ptr_to_ptr)]
97        CursorMut(unsafe {
98            Pin::new_unchecked(transmute::<
99                &mut core::mem::MaybeUninit<usize>,
100                &mut BlockHeader,
101            >(&mut self.data[0]))
102        })
103    }
104}

111impl<'a> Cursor<'a> {

113        self.0
114    }

124struct CursorMut<'a>(Pin<&'a mut BlockHeader>);
125
126impl<'a> CursorMut<'a> {
127    fn current(&mut self) -> &mut Pin<&'a mut BlockHeader> {
128        &mut self.0
129    }
130
131    fn move_next(&mut self) {
132        if !self.0.is_end() {
133            self.0 = self.0.next_mut().unwrap();
134        }
135    }
136
137    #[cfg(test)]
138    fn into_cursor(self) -> Cursor<'a> {
139        Cursor(self.0.into_ref())
140    }
141
142    fn split_exact(&mut self, new_data_size: usize) {
143        assert!(self.0.free());
144        assert!(new_data_size.is_multiple_of(align_of::<BlockHeader>()));
145
146        assert!(new_data_size <= self.0.data_size());
147        let spare_space = self.0.data_size() - new_data_size;
148        if spare_space == 0 {
149            // Nothing to do
150            return;
151        }
152
153        assert!(spare_space >= size_of::<BlockHeader>());
154        let mut spare_header = BlockHeader::default();
155        spare_header.set_free(true);
156        spare_header.set_data_size(spare_space - size_of::<BlockHeader>());
157        let spare_header_addr = self.0.as_ref().data_ptr() as usize + new_data_size;
158        let spare_header_ptr = spare_header_addr as *mut MaybeUninit<BlockHeader>;
159        unsafe {
160            *spare_header_ptr = MaybeUninit::new(spare_header);
161        }
162
163        // Update our own size last so that this is still in a safe state if anything
164        // panics above.
165        self.0.as_mut().set_data_size(new_data_size);
166    }
167
168    fn split_to_minimum(&mut self, min_data_size: usize) {
169        self.split_exact(align_up(min_data_size, Layout::new::<BlockHeader>()));
170    }
171
172    fn split_to_align(&mut self, layout: Layout) {
173        assert!(layout.align() >= align_of::<BlockHeader>());
174        let start = self.0.as_ref().data_ptr() as usize;
175        let aligned_start = align_up(start, layout);
176        if start != aligned_start {
177            let shift = aligned_start - start;
178            assert!(shift >= size_of::<BlockHeader>());
179            let new_data_size = shift - size_of::<BlockHeader>();
180            self.split_exact(new_data_size);
181            self.move_next();
182        }
183    }
184
185    fn merge_next_if_free(&mut self) {
186        match self.0.as_ref().next() {
187            Some(next) if next.free() => {
188                let additional_space = next.data_size() + size_of::<BlockHeader>();
189                let new_size = self.0.as_ref().data_size() + additional_space;
190                self.0.set_data_size(new_size);
191            }
192            _ => {}
193        };
194    }
195}

198bitfield! {
199    struct BlockHeader(usize) {
200        [0] free,
201    }
202}
203
204impl BlockHeader {
205    const SIZE_MASK: usize = !0b1;
206
207    fn data_size(self) -> usize {
208        self.0 & Self::SIZE_MASK
209    }
210
211    fn set_data_size(&mut self, value: usize) {
212        assert!(value & !Self::SIZE_MASK == 0, "Size is not 2-byte aligned");
213        self.0 &= !Self::SIZE_MASK;
214        self.0 |= value;
215    }
216
217    fn data_ptr(self: Pin<&Self>) -> *mut u8 {
218        let self_addr = unsafe { Pin::into_inner_unchecked(self) as *const _ as usize };
219        let data_addr = self_addr + size_of::<Self>();
220        data_addr as *mut u8
221    }
222
223    fn is_end(self) -> bool {
224        self.data_size() == 0 && !self.free()
225    }
226
227    fn next_ptr(self: &Pin<&Self>) -> Option<*mut Self> {
228        if self.is_end() {
229            None
230        } else {
231            let next_addr = self.data_ptr() as usize + self.data_size();
232            Some(next_addr as *mut Self)
233        }
234    }
235
236    fn next(self: Pin<&Self>) -> Option<Pin<&Self>> {
237        self.next_ptr().map(|p| unsafe { Pin::new_unchecked(&*p) })
238    }
239
240    fn next_mut<'a>(self: &Pin<&'a mut Self>) -> Option<Pin<&'a mut Self>> {
241        self.as_ref().next_ptr().map(|p| unsafe { Pin::new_unchecked(&mut *p) })
242    }
243
244    fn can_fit(self: Pin<&Self>, layout: Layout) -> bool {
245        if !self.free() {
246            return false;
247        }
248
249        // Check space without factoring in alignment
250        let self_size = self.data_size();
251        let requested_size = layout.size();
252        if requested_size > self_size {
253            return false;
254        }
255
256        // Check whether alignment is required
257        assert!(layout.align() >= align_of::<Self>());
258        let start = self.data_ptr() as usize;
259        let aligned_start = align_up(start, layout);
260        if aligned_start == start {
261            return true;
262        }
263
264        // If alignment is needed, check that the split block will still be large enough
265        let shift = aligned_start - start;
266        shift <= self_size && requested_size <= self_size - shift
267    }
268}
269
270// Several calculations in BlockHeader's methods assume this
271const_assert!(size_of::<BlockHeader>().is_multiple_of(align_of::<BlockHeader>()));

274fn align_up(addr: usize, layout: Layout) -> usize {
275    let align_mask = align_mask(layout);
276    let aligned = layout.align().wrapping_add(addr.wrapping_sub(1) & align_mask);
277    // TODO: Return Option<usize> instead?
278    assert!(
279        addr <= aligned,
280        "Aligning {:#x} to {:#x} would overflow",
281        addr,
282        layout.align()
283    );
284    aligned
285}
286
287fn align_mask(layout: Layout) -> usize {
288    // align is always a power of 2
289    !(layout.align() - 1)
290}
